function drawFinal() { if(!rawImg || !rawMask) return; const s = document.getElementById('sharpSlider').value; ctx.clearRect(0, 0, canvas.width, canvas.height); // Step 1: Create a Shrinked (Eroded) Mask to kill black edges const mCanvas = document.createElement('canvas'); mCanvas.width = canvas.width; mCanvas.height = canvas.height; const mCtx = mCanvas.getContext('2d'); // Mask ko halka sa blur karke "Contrast" barhane se edges andar shrink ho jate hain mCtx.filter = 'blur(1.5px) contrast(2) brightness(0.9)'; mCtx.drawImage(rawMask, 0, 0, canvas.width, canvas.height); // Step 2: Main Subject Rendering const temp = document.createElement('canvas'); temp.width = canvas.width; temp.height = canvas.height; const tCtx = temp.getContext('2d'); tCtx.drawImage(mCanvas, 0, 0); tCtx.globalCompositeOperation = 'source-in'; // Face & Body Sharpness tCtx.filter = `contrast(${s}) saturate(1.1) brightness(1.02)`; tCtx.drawImage(rawImg, 0, 0); // Step 3: Final Composition ctx.drawImage(temp, 0, 0); }
